$$

\cdots\

$$

1.6.3 等式

% 使用 & 使 = 左对齐

$$
\begin{aligned}
(f,K^{‘}{x}y+K^{‘’}{x}y){F}
&=(f^{‘}+f^{‘’},K^{‘}{x}y+K^{‘’}{x}y){F}\
&=(f^{‘},K^{‘}{x}y){F}+(f^{‘’},K^{‘’}{x}y){F}+(f^{‘},K^{‘’}{x}y){F}+(f^{‘’},K^{‘}{x}y){F}\
&=(f^{‘},K^{‘}{x}y){F}+(f^{‘’},K^{‘’}{x}y){F}\
&=(f^{‘}(x),y){Y}+(f^{‘’}(x),y){Y}\
&=(f^{‘}(x)+f^{‘’}(x),y){Y}\
&=(f(x),y){Y}\
&=(f,K_{x}y)_{F}
\end{aligned}
$$
